var age = prompt("Скiльки тобi рокiв?", "37");
if (+age !=NaN && age > 0) {
  if ((+age %10 == 1) && (+age %100 !=11) ){
    console.log(age,'рiк')
  }
  else if (((+age %10 >= 2) && (+age %10 <=4)) && ((+age %100 < 10) || (+age %100 >=20))){
    console.log(age,'роки')
  }
  else{
    console.log(age,'рокiв')
  }
  
}
else{
  console.log("Введіть невід'ємне число")
}
