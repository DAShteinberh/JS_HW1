console.log('number' + 3 + 3);
//Result is: "number33"
//Сперва определяем типы операндов слева-направо 
//т.к. первый операнд типа String, то последующие операнды преобразуем в String
//далее происходит последовательное слияние(конкатенация) слева-направо

console.log(null + 3);
//Result is: 3
//сначала происходит преобразование null в число, а именно в 0
//далее происходит сложение 2х чисел

console.log(5 && "qwerty");
//Result is: "qwerty"
//оператор И (&&) возвращает значение последнего операнда, если все операнды являются true
//смотрим слева направо

console.log(+'40' + +'2' + "hillel");
//Result is: "42hillel"
//сперва определяем типы операндов слева направо
//оператор унарный +, который стоит перед первым и вторым операндами с типом данных String, преобразовывает их в числа
//далее происходит сложение двух первых операндов(чисел)
//т.к тип третьего операнда String, то результат сложения первых двух операндов преобразовываем также в String
//далее происходит слияние(конкатенация) слева направо

console.log('10' - 5 === 6);
//Result is: false
//вычитание имеет выше приоритет выполнения, чем строгое равенство, поэтому начинаем с него
//для этого преобразовываем первый операнд с типом данных String в число
//далее выполняем вычитание
//далее выполняем строгое равенство, результатом которого будет false

console.log(true + false);
//Result is: 1
//сперва определяем типы операндов слева направо
//далее преобразовываем оба операнда, тип которых Boolean, в число, а именно true в 1, а false в 0
//далее происходит сложение двух чисел

console.log('4px' - 3);
//Result is: NaN
//сперва преобразовываем первый операнд с типом String в число
//в случае с вычитанием результатом преобразования будет NaN
//т.к. один из операндов вычитания является NaN, то и результатом будет NaN

console.log('4' - 3);
//Result is: 1
//сперва определяем типы операндов слева направо
//далее преобразовываем первый операнд, тип которого String, в число
//далее происходит вычитание

console.log('6' + 3 ** 0);
//Result is: "61"
//сперва выполняем возведение в степень(**)
//далее результат возведения в степень преобразовываем в String, т. к. первый операнд имеет тип String
//далее происходит слияние(конкатенация) слева направо

console.log(12/'6');
//Result is: 2
//сперва определяем типы операндов слева направо
//далее преобразовываем второй операнд, тип которого String, в число
//далее происходит деление

console.log('10' + (5 === 6));
//Result is: "10false"
//сперва выполняем действие в скобках, а именно строгое равенство(===), результатом которого будет false
//далее преобразовываем false в String, т. к. первый операнд имеет тип String
//далее происходит слияние(конкатенация) слева направо

console.log(null == '');
//Result is: false
//в случае с оператором Равенства(==), если один из операторов null, то другой должен быть также null или undefined, чтобы вернулось значение true. В противном случае возвращается false.

console.log(3 ** (9 / 3));
//Result is: 27
//сперва выполняем действие в скобках, а именно деление(/)
//далее возводим первый операнд в степень, полученную в результате деления

console.log(!!'false' == !!'true');
//Result is: true
//сперва выполняем двойное НЕ(!!), а далее Равенство(==)
//при выполнении действий с логическими операторами любая не пустая Stringа принимает значение true
//исходя из этого, 'false' это true, тогда !!true будет тоже true (одно отрицание преобразует в false, а второе отрицание преобразует в true)
//также 'true' является не пустой Stringой и принимает значение true и, соответственно, !!true будет тоже true
//получаем следующее выражение true == true, результатом которого является true

console.log(0 || '0' && 1);
//Result is: 1
//оператор И (&&) имеет выше приоритет выполнения, чем оператор ИЛИ (||), поэтому начинаем с него
//при выполнении действий с оператором И (&&) любая не пустая Stringа принимает значение true
//далее оператор И (&&) возвращает значение последнего операнда, если все операнды являются true, то есть 1
//далее оператор ИЛИ (||) возващает то значение, которое является правдивым, а это 1

console.log((+null == false) < 1);
//Result is: false
//сперва выполняем действие в скобках, а именно равенство(==)
//оператор унарный +, который стоит перед null, преобразовывает его в число, а именно в 0
//результатом выполнения равенства будет true
//далее переходим к сравнению, а для этого преобразовываем полученный результат(true) в число, а именно в 1
//далее выполняем сравнение 1 < 1, что является false

console.log(false && true || true);
//Result is: true
//оператор И (&&) имеет выше приоритет выполнения, чем оператор ИЛИ (||), поэтому начинаем с него
//оператор И (&&) возващает false, если хотя бы один из операндов не является true
//далее выполняется оператор ИЛИ (||), который возващает true, если хотя бы один из операндов является true

console.log(false && (false || true));
//Result is: false
//сперва выполняем действие в скобках
//оператор ИЛИ (||) возващает true, если хотя бы один из операндов является true
//далее выполняется оператор И (&&), который возващает false, если хотя бы один из операндов не является true

console.log((+null == false) < 1 ** 5);
//Result is: false
//сперва выполняем действие в скобках, а именно равенство(==)
//оператор унарный +, который стоит перед null, преобразовывает его в число, а именно в 0
//результатом выполнения действия в скобках будет true
//далее выполняем возведение в степень(**), результатом которого является 1
//далее переходим к сравнению, а для этого преобразовываем полученный результат действий в скобках(true) в число, а именно в 1
//далее выполняем сравнение 1 < 1, что является false
